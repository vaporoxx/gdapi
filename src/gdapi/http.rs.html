44#[derive(Debug, Default)]
45pub struct Http {
46	auth: OnceLock<Auth>,
47	client: Client,
48}
49
50impl Http {
51	const API_BASE: &str = "https://www.boomlings.com/database";
52
53	pub fn auth(&self) -> Result<&Auth> {
54		self.auth.get().ok_or(Error::NotLoggedIn)
55	}
56
57	pub fn authenticate(&self, account_id: AccountId, gjp2: String) -> Result<()> {
58		self.auth.set(Auth { account_id, gjp2 }).or(Err(Error::AlreadyLoggedIn))
59	}
60
61	pub async fn post<T: Parse>(&self, endpoint: Endpoint, form: impl Serialize) -> Result<T> {
62		let url = format!("{}/{}.php", Self::API_BASE, endpoint.endpoint());
63
64		let response = self.client.post(url).form(&form).send().await?;
65		let data = response.error_for_status()?.text().await?;
66
67		if data.starts_with('-') {
68			return Err(Error::InvalidRequest(data.parse().unwrap_or_default()));
69		}
70
71		if let Some((data, remaining)) = data.split_once('#') {
72			T::parse(data, Some(remaining))
73		} else {
74			T::parse(&data, None)
75		}
76	}
