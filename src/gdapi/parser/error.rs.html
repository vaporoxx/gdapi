1use std::num::ParseIntError;
2use std::{error, fmt};
3
4/// The error type for all parsing actions.
5#[derive(Debug)]
6pub enum Error {
7	/// An enum value is out of range
8	InvalidEnumValue(u8),
9	/// A required key is not present
10	InvalidKey(u8),
11	/// An array string has an invalid length
12	InvalidLength,
13	/// A key-value string has an odd length
14	OddElements,
15	/// A wrapper around a [`ParseIntError`]
16	ParseInt(ParseIntError),
17}
18
19impl fmt::Display for Error {
20	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
21		match self {
22			Self::InvalidEnumValue(key) => write!(f, "enum value out of range (key {key})"),
23			Self::InvalidKey(key) => write!(f, "required key not present (key {key})"),
24			Self::InvalidLength => write!(f, "array string has invalid length"),
25			Self::OddElements => write!(f, "key-value string has odd length"),
26			Self::ParseInt(error) => error.fmt(f),
27		}
28	}
29}
30
31impl error::Error for Error {
32	fn source(&self) -> Option<&(dyn error::Error + 'static)> {
33		match self {
34			Self::InvalidEnumValue(_) => None,
35			Self::InvalidKey(_) => None,
36			Self::InvalidLength => None,
37			Self::OddElements => None,
38			Self::ParseInt(error) => Some(error),
39		}
40	}
41}
42
43impl From<ParseIntError> for Error {
44	fn from(error: ParseIntError) -> Self {
45		Self::ParseInt(error)
46	}
47}
